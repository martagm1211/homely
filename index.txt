window.switchAuthTab = (tab) => {
  document.getElementById('login-form').style.display = tab === 'login' ? 'block' : 'none';
  document.getElementById('register-form').style.display = tab === 'register' ? 'block' : 'none';
  document.getElementById('tab-login').classList.toggle('active', tab === 'login');
  document.getElementById('tab-register').classList.toggle('active', tab === 'register');
  document.getElementById('auth-msg').textContent = '';
};

window.doLogin = async () => {
  const email = document.getElementById('login-email').value.trim();
  const pass = document.getElementById('login-pass').value;
  try {
    await signInWithEmailAndPassword(auth, email, pass);
  } catch(e) {
    document.getElementById('auth-msg').textContent = lang === 'es' ? 'Correo o contraseña incorrectos' : 'Incorrect email or password';
  }
};

window.doRegister = async () => {
  const name = document.getElementById('reg-name').value.trim();
  const email = document.getElementById('reg-email').value.trim();
  const pass = document.getElementById('reg-pass').value;
  if (!name) { document.getElementById('auth-msg').textContent = lang === 'es' ? 'Escribe tu nombre' : 'Enter your name'; return; }
  try {
    const cred = await createUserWithEmailAndPassword(auth, email, pass);
    await updateProfile(cred.user, { displayName: name });
    await setDoc(doc(db, 'users', cred.user.uid), { name, email, uid: cred.user.uid, createdAt: serverTimestamp() });
  } catch(e) {
    document.getElementById('auth-msg').textContent = e.message;
  }
};

window.doLogout = async () => {
  unsubAll();
  currentHouseId = null;
  currentHouseData = null;
  await signOut(auth);
};

});

async function loadHouses() {
  const list = document.getElementById('house-list');
  list.innerHTML = '<div class="loading"><div class="loading-spinner"></div></div>';

  // My houses
  const q = query(collection(db, 'houses'), where('memberIds', 'array-contains', currentUser.uid));
  onSnapshot(q, (snap) => {
    const houses = [];
    snap.forEach(d => houses.push({ id: d.id, ...d.data() }));

    // Pending requests
    const q2 = query(collection(db, 'houses'), where('pendingIds', 'array-contains', currentUser.uid));
    onSnapshot(q2, (snap2) => {
      const pending = [];
      snap2.forEach(d => pending.push({ id: d.id, ...d.data() }));
      renderHouseList(houses, pending);
    });
  });
}

function renderHouseList(houses, pending) {
  const list = document.getElementById('house-list');
  let html = '';
  houses.forEach(h => {
    const isAdmin = h.adminId === currentUser.uid;
    html += `<div class="house-card" onclick="enterHouse('${h.id}')">
      <div class="house-icon">🏠</div>
      <div class="house-card-info"><h3>${h.name}</h3><p>${h.memberIds?.length || 1} ${lang==='es'?'miembro(s)':'member(s)'}</p></div>
      ${isAdmin ? `<div class="house-badge">${t('adminBadge')}</div>` : ''}
    </div>`;
  });
  pending.forEach(h => {
    html += `<div class="house-card" style="opacity:.6;cursor:default">
      <div class="house-icon">⏳</div>
      <div class="house-card-info"><h3>${h.name}</h3><p>${lang==='es'?'Esperando aprobación':'Awaiting approval'}</p></div>
      <div class="house-badge pending">${t('pendingBadge')}</div>
    </div>`;
  });
  html += `<button class="add-house-btn" onclick="openModal('modal-create-house')">🏡 <span>${lang==='es'?'Crear nueva casa':'Create new home'}</span></button>`;
  html += `<button class="add-house-btn" style="margin-top:8px" onclick="openModal('modal-join-house')">🔑 <span>${lang==='es'?'Unirse a una casa':'Join a home'}</span></button>`;
  list.innerHTML = html;
}

window.enterHouse = async (houseId) => {
  currentHouseId = houseId;
  const snap = await getDoc(doc(db, 'houses', houseId));
  currentHouseData = { id: houseId, ...snap.data() };
  document.getElementById('current-house-name').textContent = currentHouseData.name;
  document.getElementById('current-user-label').textContent = currentUser.displayName || currentUser.email;
  showScreen('app');
  switchTab('calendar');
  subscribeToHouse();
};

window.goToHouses = () => {
  unsubAll();
  showScreen('house');
  loadHouses();
};

window.createHouse = async () => {
  const name = document.getElementById('new-house-name').value.trim();
  if (!name) return;
  const ref = await addDoc(collection(db, 'houses'), {
    name, adminId: currentUser.uid,
    memberIds: [currentUser.uid],
    pendingIds: [],
    kids: [], extraMeds: [],
    createdAt: serverTimestamp()
  });
  await setDoc(doc(db, 'houses', ref.id, 'members', currentUser.uid), {
    uid: currentUser.uid, name: currentUser.displayName || currentUser.email,
    email: currentUser.email, role: 'admin'
  });
  closeModal('modal-create-house');
  showToast(t('houseCreated'));
  enterHouse(ref.id);
};

window.requestJoin = async () => {
  const hid = document.getElementById('join-house-id').value.trim();
  if (!hid) return;
  const ref = doc(db, 'houses', hid);
  const snap = await getDoc(ref);
  if (!snap.exists()) { showToast(lang==='es'?'Casa no encontrada':'Home not found'); return; }
  await updateDoc(ref, { pendingIds: arrayUnion(currentUser.uid) });
  // Save requester info
  await setDoc(doc(db, 'houses', hid, 'requests', currentUser.uid), {
    uid: currentUser.uid, name: currentUser.displayName || currentUser.email,
    email: currentUser.email, requestedAt: serverTimestamp()
  });
  closeModal('modal-join-house');
  showToast(t('requestSent'));
};

// ── SUBSCRIBE TO HOUSE DATA ──────────────────────
function subscribeToHouse() {
  unsubAll();

  // Members
  const uMembers = onSnapshot(collection(db, 'houses', currentHouseId, 'members'), snap => {
    allMembers = [];
    snap.forEach(d => allMembers.push(d.data()));
    renderMembersConfig();
    renderTaskAssignSelect();
  });

  // House doc (kids, extraMeds)
  const uHouse = onSnapshot(doc(db, 'houses', currentHouseId), snap => {
    currentHouseData = { id: currentHouseId, ...snap.data() };
    allKids = currentHouseData.kids || [];
    extraMeds = currentHouseData.extraMeds || [];
    document.getElementById('house-name-input').value = currentHouseData.name || '';
    renderKidsTable();
    renderMedsTable();
    renderKidsConfig();
    renderEvMembersChips();
    renderPendingRequests();
  });

  // Events
  const uEvents = onSnapshot(query(collection(db, 'houses', currentHouseId, 'events'), orderBy('date')), snap => {
    allEvents = [];
    snap.forEach(d => allEvents.push({ id: d.id, ...d.data() }));
    renderCalendar();
    renderEventsForDay(selectedCalDay);
  });

  // Tasks
  const uTasks = onSnapshot(collection(db, 'houses', currentHouseId, 'tasks'), snap => {
    allTasks = [];
    snap.forEach(d => allTasks.push({ id: d.id, ...d.data() }));
    renderTasks();
  });

  // Recipes
  const uRecipes = onSnapshot(collection(db, 'houses', currentHouseId, 'recipes'), snap => {
    allRecipes = [];
    snap.forEach(d => allRecipes.push({ id: d.id, ...d.data() }));
    renderRecipes();
  });

  // Weekly menus
  renderWeekMenu();

  unsubListeners = [uMembers, uHouse, uEvents, uTasks, uRecipes];

  // Add self as member if not yet
  setTimeout(async () => {
    const memRef = doc(db, 'houses', currentHouseId, 'members', currentUser.uid);
    const memSnap = await getDoc(memRef);
    if (!memSnap.exists()) {
      await setDoc(memRef, {
        uid: currentUser.uid, name: currentUser.displayName || currentUser.email,
        email: currentUser.email,
        role: currentHouseData?.adminId === currentUser.uid ? 'admin' : 'member'
      });
    }
  }, 500);
}

function unsubAll() { unsubListeners.forEach(u => u && u()); unsubListeners = []; }

// ── CALENDAR ────────────────────────────────────
function renderCalendar() {
  const grid = document.getElementById('cal-grid');
  const year = calDate.getFullYear();
  const month = calDate.getMonth();
  document.getElementById('cal-month-label').textContent = `${t('months')[month]} ${year}`;

  const dows = t('dowNames');
  let html = dows.map(d => `<div class="cal-dow">${d}</div>`).join('');

  const firstDay = new Date(year, month, 1);
  let startDow = firstDay.getDay(); // 0=Sun
  startDow = startDow === 0 ? 6 : startDow - 1; // Mon=0
  const daysInMonth = new Date(year, month+1, 0).getDate();
  const prevDays = new Date(year, month, 0).getDate();

  const today = new Date();
  const eventDays = new Set(allEvents.filter(e => {
    const d = new Date(e.date);
    return d.getFullYear() === year && d.getMonth() === month;
  }).map(e => new Date(e.date).getDate()));

  // Prev month
  for (let i = startDow-1; i >= 0; i--) {
    html += `<div class="cal-day other-month">${prevDays-i}</div>`;
  }
  // Current month
  for (let d = 1; d <= daysInMonth; d++) {
    const isToday = today.getFullYear()===year && today.getMonth()===month && today.getDate()===d;
    const hasEv = eventDays.has(d);
    const dateStr = `${year}-${String(month+1).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
    html += `<div class="cal-day${isToday?' today':''}${hasEv?' has-event':''}" onclick="selectDay('${dateStr}')">${d}</div>`;
  }
  // Next month
  const total = startDow + daysInMonth;
  const remaining = total % 7 === 0 ? 0 : 7 - (total % 7);
  for (let d = 1; d <= remaining; d++) {
    html += `<div class="cal-day other-month">${d}</div>`;
  }
  grid.innerHTML = html;
}

window.changeMonth = (dir) => {
  calDate = new Date(calDate.getFullYear(), calDate.getMonth() + dir, 1);
  renderCalendar();
};

function renderEventsForDay(dateStr) {
  const container = document.getElementById('events-for-day');
  if (!dateStr) { container.innerHTML = ''; return; }
  const dayEvents = allEvents.filter(e => e.date === dateStr);
  if (!dayEvents.length) {
    container.innerHTML = `<div class="empty-state"><p>${t('noEvents')}</p></div>`;
    return;
  }
  const evTypes = t('evTypes');
  const colors = { medico:'#C0392B', celebracion:'#E67E22', escolar:'#2980B9', extraescolar:'#4A7C59', evento:'#8E44AD', otros:'#888' };
  container.innerHTML = dayEvents.map(ev => `
    <div class="event-card">
      <div class="event-dot" style="background:${colors[ev.type]||'#888'}"></div>
      <div class="event-info">
        <h4>${ev.title}</h4>
        <p>${ev.members?.join(', ') || ''}</p>
        <span class="event-tag ev-${ev.type}">${evTypes[ev.type]||ev.type}</span>
      </div>
      <button style="background:none;border:none;cursor:pointer;font-size:18px;color:var(--text-soft)" onclick="deleteEvent('${ev.id}')">🗑️</button>
    </div>`).join('');
}

function getWeekStart(offset) {
  const now = new Date();
  const day = now.getDay();
  const diff = now.getDate() - (day === 0 ? 6 : day - 1);
  const monday = new Date(now.setDate(diff + offset * 7));
  monday.setHours(0,0,0,0);
  return monday;
}

function renderWeekMenu() {
  const weekStart = getWeekStart(currentWeekOffset);
  const weekEnd = new Date(weekStart); weekEnd.setDate(weekEnd.getDate() + 6);
  const fmt = d => `${String(d.getDate()).padStart(2,'0')}/${String(d.getMonth()+1).padStart(2,'0')}`;
  document.getElementById('week-label').textContent = `${t('weekOf')} ${fmt(weekStart)} – ${fmt(weekEnd)}`;

  const weekKey = `${weekStart.getFullYear()}-W${String(Math.ceil(weekStart.getDate()/7)).padStart(2,'0')}-${currentWeekOffset}`;
  const dayNames = t('days');
  const lunch = t('lunch'); const dinner = t('dinner');
  let html = '';

  for (let i = 0; i < 7; i++) {
    const d = new Date(weekStart); d.setDate(d.getDate() + i);
    const dateStr = d.toISOString().split('T')[0];
    const dayKey = `${weekKey}-${dateStr}`;
    const saved = currentHouseData?.menus?.[dateStr] || {};
    html += `<div class="day-menu-card">
      <div class="day-menu-header">${dayNames[i]} ${fmt(d)}</div>
      <div class="meal-row"><span class="meal-label">${lunch}</span><input type="text" value="${saved.lunch||''}" placeholder="..." onblur="saveMenu('${dateStr}','lunch',this.value)"></div>
      <div class="meal-row"><span class="meal-label">${dinner}</span><input type="text" value="${saved.dinner||''}" placeholder="..." onblur="saveMenu('${dateStr}','dinner',this.value)"></div>
    </div>`;
  }
  document.getElementById('week-days').innerHTML = html;
}

function renderTaskAssignSelect() {

function renderTasks() {
  const pending = allTasks.filter(t => !t.done);
  const done = allTasks.filter(t => t.done);

  const pContainer = document.getElementById('tasks-pending');
  const dContainer = document.getElementById('tasks-done');

  if (!pending.length) {
    pContainer.innerHTML = `<div class="empty-state"><div class="empty-icon">🎉</div><p>${t('noTasks')}</p></div>`;
  } else {
    pContainer.innerHTML = pending.map(task => renderTaskCard(task)).join('');
  }

  if (!done.length) {
    dContainer.innerHTML = `<div class="empty-state" style="padding:16px 0"><p style="font-size:13px">${t('noDoneTasks')}</p></div>`;
  } else {
    dContainer.innerHTML = done.map(task => renderTaskCard(task)).join('');
  }
}

function renderTaskCard(task) {
  const assignee = task.assignedTo ? allMembers.find(m=>m.uid===task.assignedTo)?.name || '' : '';
  const doneBy = task.doneBy ? allMembers.find(m=>m.uid===task.doneBy)?.name || '' : '';
  const doneAt = task.doneAt?.toDate ? task.doneAt.toDate().toLocaleString(lang==='es'?'es-ES':'en-GB', {dateStyle:'short',timeStyle:'short'}) : '';
  return `<div class="task-item">
    <div class="task-check${task.done?' done':''}" onclick="toggleTask('${task.id}',${!task.done})">${task.done?'✓':''}</div>
    <div class="task-info">
      <h4 class="${task.done?'done-text':''}">${task.title}</h4>
      ${assignee ? `<p class="task-meta">${t('assignedTo')}: ${assignee}</p>` : ''}
      ${task.done && doneBy ? `<p class="task-done-by">✓ ${t('doneBy')} ${doneBy} · ${doneAt}</p>` : ''}
    </div>
    <button style="background:none;border:none;cursor:pointer;font-size:16px;color:var(--text-soft)" onclick="deleteTask('${task.id}')">🗑️</button>
  </div>`;
}

function renderPendingRequests() {
  const container = document.getElementById('pending-requests');
  const isAdmin = currentHouseData?.adminId === currentUser.uid;
  if (!isAdmin) { container.innerHTML = ''; return; }

  const pendingIds = currentHouseData.pendingIds || [];
  if (!pendingIds.length) { container.innerHTML = ''; return; }

  // Fetch request details
  Promise.all(pendingIds.map(uid => getDoc(doc(db, 'houses', currentHouseId, 'requests', uid))))
    .then(snaps => {
      const requests = snaps.filter(s=>s.exists()).map(s=>s.data());
      if (!requests.length) { container.innerHTML = ''; return; }
      container.innerHTML = `<div style="font-size:13px;font-weight:700;color:var(--amber);margin:12px 0 8px">${t('pendingRequests')}</div>` +
        requests.map(r => `<div class="request-card">
          <div class="avatar" style="background:var(--amber)">${(r.name||'?').charAt(0).toUpperCase()}</div>
          <div style="flex:1"><strong style="font-size:14px">${r.name}</strong><br><span style="font-size:12px;color:var(--text-soft)">${r.email}</span></div>
          <div class="request-actions">
            <button class="btn-accept" onclick="acceptRequest('${r.uid}','${r.name}','${r.email}')">${t('acceptBtn')}</button>
            <button class="btn-reject" onclick="rejectRequest('${r.uid}')">${t('rejectBtn')}</button>
          </div>
        </div>`).join('');
    });
}

window.acceptRequest = async (uid, name, email) => {
  const houseRef = doc(db, 'houses', currentHouseId);
  // Add to members
  await setDoc(doc(db, 'houses', currentHouseId, 'members', uid), { uid, name, email, role: 'member' });
  // Add to memberIds, remove from pending
  const h = currentHouseData;
  const newPending = (h.pendingIds||[]).filter(id=>id!==uid);
  const newMembers = [...(h.memberIds||[]), uid];
  await updateDoc(houseRef, { memberIds: newMembers, pendingIds: newPending });
  await deleteDoc(doc(db, 'houses', currentHouseId, 'requests', uid));
  showToast(t('accepted'));
};

window.rejectRequest = async (uid) => {
  const h = currentHouseData;
  const newPending = (h.pendingIds||[]).filter(id=>id!==uid);
  await updateDoc(doc(db, 'houses', currentHouseId), { pendingIds: newPending });
  await deleteDoc(doc(db, 'houses', currentHouseId, 'requests', uid));
  showToast(t('rejected'));
};
